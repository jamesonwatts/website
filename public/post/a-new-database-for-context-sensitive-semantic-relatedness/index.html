import pandas as pd
import chardet
import numpy as np

    result = chardet.detect(f.read())  # or readline if the file is large
dfi = pd.read_csv(f"{homedir}/private/master.csv", encoding=result['encoding'])
## sys:1: DtypeWarning: Columns (22,23) have mixed types. Specify dtype option on import or set low_memory=False.
#agreement stats
taskdata = {"ENTERTAINMENT":{},"TECHNOLOGY":{},"POLITICS":{}}
for index, row in dfi.iterrows():
    for j in range(50):
        compare = row["Input.Pairwise{:02}-P1".format(j+1)]+"~"+row["Input.Pairwise{:02}-P2".format(j+1)]
        answer = row["Answer.Pairwise{:02}".format(j+1)]
        context = row["Input.Context"]
        if compare in taskdata[context]:
            taskdata[context][compare].append(answer) 
        else:
            taskdata[context][compare]=[answer]
#accumulate scores
pairs = {}
totalerrors=0
for index, row in dfi.iterrows(): 
    for j in range(50):
        p1 = row["Input.Pairwise{:02}-P1".format(j+1)]
        p2 = row["Input.Pairwise{:02}-P2".format(j+1)]
        compare = p1+"~"+p2
        context = row["Input.Context"]
        judgements = np.mean(taskdata[context][compare])  
        
        if p1 not in pairs.keys():
            pairs[p1] = {"ENTERTAINMENT":[],"TECHNOLOGY":[],"POLITICS":[],"W_ENTERTAINMENT":[],"W_TECHNOLOGY":[],"W_POLITICS":[],"A_ENTERTAINMENT":[],"A_TECHNOLOGY":[],"A_POLITICS":[],"ALL":[],"AGE":[],"EDU":[],"ETH":[],"GEN":[],"US":[]}
        
        answer = row["Answer.Pairwise{:02}".format(j+1)]
        weight = (row['Answer.Familiarity']/2)-0.5 #1->0, 2->0.5, 3->1
        pairs[p1][context].append(1-answer) #0 means focus pair is more related.
        pairs[p1][f"W_{context}"].append((1-answer)*weight) #0 means focus pair is more related.
        
        pairs[p1][f"A_{context}"].append(1 if judgements == 0 or judgements == 1 else 0) #agree if value is either 0 or 2
        pairs[p1]['ALL'].append(1-answer) #0 means focus pair is more related.
        pairs[p1]['AGE'].append(row['Answer.Age'])
        pairs[p1]['EDU'].append(row['Answer.Education'])
        pairs[p1]['ETH'].append(1 if row['Answer.Ethnicity']=='White' else 0)
        pairs[p1]['GEN'].append(1 if row['Answer.Gender']=='Male' else 0)
        pairs[p1]['US'].append(1 if row['Answer.country']=='US' else 0)
                        
dfi.to_csv(f"{homedir}/master-results.csv")                
print("finished aggregation")
